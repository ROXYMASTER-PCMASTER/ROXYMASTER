[ALTO #2] Autenticación básica en WebSocket
────────────────────────────────────────────
NUEVAS VARIABLES (línea 42-43):
  tokens_autenticados = {}  # token -> {cid, tiempo}
  TOKEN_TIMEOUT = 3600  # 1 hora

NUEVAS FUNCIONES (línea 57-75):
  
  def generar_token():
      """Genera un token seguro de autenticación"""
      return secrets.token_urlsafe(32)
  
  def verificar_token(token):
      """Verifica si el token es válido y no ha expirado"""
      if token not in tokens_autenticados:
          return None
      
      token_info = tokens_autenticados[token]
      if time.time() - token_info["tiempo"] > TOKEN_TIMEOUT:
          del tokens_autenticados[token]
          return None
      
      return token_info["cid"]

IMPORTACIÓN AÑADIDA (línea 14):
  import secrets

FLUJO DE AUTENTICACIÓN:
  1. PCBOT se conecta con "identify"
  2. Servidor genera token único: generar_token()
  3. Token se almacena con CID y timestamp
  4. Token se envía a PCBOT en respuesta "connected"
  5. PCBOT puede usar token para futuras operaciones (opcional)
  6. Token expira después de 1 hora

CAMBIO EN manejar_conexion() (línea 185-195):

  async def enviar(pcbot_id, cmd, data):
      try:
          with lock_pcbots:
              if pcbot_id not in pcbots:
                  print(f"[ADVERTENCIA] PCBOT {pcbot_id} no conectado")
                  return False
              ws = pcbots[pcbot_id]
          
          try:
              msg = json.dumps({"type": cmd, "data": data})
              await ws.send(msg)
              return True
          except websockets.exceptions.ConnectionClosed:
              print(f"[-] PCBOT {pcbot_id} desconectado")
              with lock_pcbots:
                  pcbots.pop(pcbot_id, None)
              return False
      except json.JSONEncodeError as e:
          print(f"[ERROR] JSON inválido: {cmd} - {e}")
          return False
      except Exception as e:
          print(f"[ERROR] Fallo enviando: {e}")
          return False
